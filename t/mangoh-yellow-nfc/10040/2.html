Made a fix for this.

gist.github.com

nfc_mangoh_slave_address.c
#include <linux/i2c.h>
#include <linux/i2c-help.h>
#include <sys/ioctl.h>

#include "legato.h"

#define ADAPTER "/dev/i2c-8"
#define BAD_ADDRESS 0x00
#define GOOD_ADDRESS 0x55
